step 1 - Get access token

TOKEN=$(curl -s -X POST "https://champ.jamfcloud.com/api/v1/oauth/token" \
  -H "Content-Type: application/x-www-form-urlencoded" \
  -d "client_id=(client id here)&client_secret=(secret here)&grant_type=client_credentials" \
  | jq -r '.access_token')

step 2 - optional 

echo $TOKEN (to see the access token

step 3 - get list of all computers

curl -X GET "https://champ.jamfcloud.com/api/v1/computers-inventory" \
  -H "Authorization: Bearer $TOKEN" \
  -H "Accept: application/json" | jq

step 4 list of all policies

curl -X GET "https://champ.jamfcloud.com/JSSResource/policies" \
  -H "Authorization: Bearer $TOKEN" \
  -H "Accept: application/json" | jq
